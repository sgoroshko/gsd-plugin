<!-- Public contract this module MUST expose. Plan 11-04 (cmdVerifyDrift) calls ONLY detect(). -->

detect(corpus, opts?) -> {
  skipped: false,
  pairs: [ { a: {file, name, startLine}, b: {file, name, startLine}, similarity } ],
  suppressed: [ { a, b, similarity, reason } ]
} | { skipped: true, reason, pairs: [], suppressed: [] }

corpus = string[] of cwd-relative file paths (from collectConventionCorpus; the module reads each).
opts.cwd = base dir for reads. opts.allow = { pairs: [{a,b,reason}], ignore: [] } from drift-allowlist.cjs (Plan 11-02). When opts.allow is absent, suppressed:[] is empty.

Ported constants (verbatim from @vibedrift/cli@0.14.4, RESEARCH §Code Examples — these ARE the answers to the "tune the params" discretion item):
  DEFAULT_SHINGLE_SIZE = 5, DEFAULT_PERMUTATIONS = 128, DEFAULT_LSH_BANDS = 16,
  DEFAULT_LSH_ROWS = 8 (16x8 = 128), MIN_BODY_TOKENS = 15, FLAG_THRESHOLD = 0.7.
</interfaces>
